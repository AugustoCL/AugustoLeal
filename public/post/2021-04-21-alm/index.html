SIMPLEX Model

m = JuMP.Model(Cbc.Optimizer);
set_silent(m)
true

@variable(m, x[1:N] >= 0, Int);
@variable(m, f[1:Y] >= 0);

@constraint(m, con1,    x'*C[1,:]          == cash_flow[1] + f[1]);
@constraint(m, con[i=2:Y], x'*C[i,:] + f[i-1] == cash_flow[i] + f[i]);

@objective(m, Min, x'*bond_prices);

JuMP.optimize!(m);
